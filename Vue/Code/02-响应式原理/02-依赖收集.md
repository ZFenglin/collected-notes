# 依赖收集

之前响应式中提到过，在响应式属性触发get时会进行依赖收集

而render函数执行时，就会触发模板中属性的get，从而进行收集

而依赖收集主要时由Watcher和Dep这两个类负责处理

## Dep

与响应式处理和Observer对象共同创建，一个响应式对象对应一个dep

```js
export default class Dep {
    // 内部属性声明
    //...
    constructor() {
        this.id = uid++
        this.subs = []
    }
    // dep收集传入watcher（wather的addDep触发）
    addSub(sub) {
        this.subs.push(sub)
    }
    // 移除传入的watcher
    removeSub(sub) {
        remove(this.subs, sub)
    }
    // 依赖收集触发
    depend() {
        if (Dep.target) {
            Dep.target.addDep(this)
        }
    }
    // 通知watcher更新
    notify() {
        const subs = this.subs.slice()
        // 遍历调用sub的watcher的update
        for (let i = 0, l = subs.length; i < l; i++) {
            subs[i].update()
        }
    }
}
```

## Dep.target处理

Dep.target设置当前全局的watcher

targetStack维护处理的watcher栈

```js
Dep.target = null
const targetStack = []
export function pushTarget(target) {
    targetStack.push(target)
    Dep.target = target
}

export function popTarget() {
    targetStack.pop()
    Dep.target = targetStack[targetStack.length - 1]
}
```

## Watcher

constructor执行流程
1. 设置options
2. 其他属性设置

```js
export default class Watcher {
    // 内部属性声明
    // ...
    constructor(
        vm,
        expOrFn,
        cb,
        options,
        isRenderWatcher
    ) {
        this.vm = vm
        if (isRenderWatcher) {
            vm._watcher = this
        }
        vm._watchers.push(this) // 强制更新$forceUpdate使用
        // 1. 设置options
        if (options) {
            this.deep = !!options.deep // 深度监听
            this.user = !!options.user // 用户定义watcher
            this.lazy = !!options.lazy // 懒加载，computed
            this.sync = !!options.sync // 同步执行，内部使用
            this.before = options.before // 更新前触发的钩子
        } else {
            this.deep = this.user = this.lazy = this.sync = false
        }
        // 2. 其他属性设置
        this.cb = cb // 回调函数
        this.id = ++uid // uid
        this.active = true // 激活标识
        this.dirty = this.lazy // 懒加载脏数据标识
        // dep收集
        this.deps = []
        this.newDeps = []
        this.depIds = new Set()
        this.newDepIds = new Set()
        this.expression = ''
        // 获取实际执行的expOrFn
        // 按照watcher的不同，expOrFn可能是方法或者访问对象路径
        if (typeof expOrFn === 'function') {
            this.getter = expOrFn
        } else {
            // 自定义watch的对象路径
            this.getter = parsePath(expOrFn)
            if (!this.getter) {
                this.getter = noop
            }
        }
        this.value = this.lazy ?
            undefined :
            this.get()
    }
    // 触发依赖收集和值更新处理
    get() {
        // 将当前的watcher设置成目标
        pushTarget(this)
        let value
        const vm = this.vm
        try {
            // 执行get，在触发响应式的get时会收集当前的watcher
            value = this.getter.call(vm, vm)
        } catch (e) {
            if (this.user) {
                handleError(e, vm, `getter for watcher "${this.expression}"`)
            } else {
                throw e
            }
        } finally {
            if (this.deep) {
                // 深度监听处理
                traverse(value)
            }
            // 清除目标
            popTarget()
            this.cleanupDeps() /// 不清理， 可能上次的数据会被再次收集
        }
        return value
    }
    // 添加对应dep
    addDep(dep: Dep) {
        const id = dep.id
        if (!this.newDepIds.has(id)) {
            this.newDepIds.add(id)
            this.newDeps.push(dep)
            if (!this.depIds.has(id)) {
                dep.addSub(this)
            }
        }
    }
    // 清理deps收集
    cleanupDeps() {
        let i = this.deps.length
        while (i--) {
            const dep = this.deps[i]
            if (!this.newDepIds.has(dep.id)) {
                dep.removeSub(this)
            }
        }
        let tmp = this.depIds
        this.depIds = this.newDepIds
        this.newDepIds = tmp
        this.newDepIds.clear()
        tmp = this.deps
        this.deps = this.newDeps
        this.newDeps = tmp
        this.newDeps.length = 0
    }
    // watcher更新处理
    update() {
        if (this.lazy) {
            // computed的watcher将数据设置为脏数据
            this.dirty = true
        } else if (this.sync) {
            this.run() // 开发环境下使用不到
        } else {
            // 当前watch入队，等待异步更新处理
            queueWatcher(this)
        }
    }
    // 执行get和回调处理（scheduler处理时调用执行）
    run() {
        if (this.active) {
            const value = this.get()
            if (
                value !== this.value ||
                isObject(value) ||
                this.deep
            ) {
                // 新旧值处理
                const oldValue = this.value
                this.value = value
                // 回调触发
                if (this.user) {
                    const info = `callback for watcher "${this.expression}"`
                    invokeWithErrorHandling(this.cb, this.vm, [value, oldValue], this.vm, info)
                } else {
                    this.cb.call(this.vm, value, oldValue)
                }
            }
        }
    }
    // computed懒加载值更新处理
    evaluate() {
        this.value = this.get()
        this.dirty = false
    }
    // 触发deps中所有的dep收集当前watcher
    depend() {
        let i = this.deps.length
        while (i--) {
            this.deps[i].depend()
        }
    }
    // 当前watcher销毁
    teardown() {
        if (this.active) {
            // 1. vm._watchers移除当前watcher
            if (!this.vm._isBeingDestroyed) {
                remove(this.vm._watchers, this)
            }
            // 2. deps去除当前watcher
            let i = this.deps.length
            while (i--) {
                this.deps[i].removeSub(this)
            }
            this.active = false
        }
    }
}
```
